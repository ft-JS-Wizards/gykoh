function solution(n, control) {
  let ans = n;
  for (let i = 0; i < control.length; i++) {
    if (control[i] === "w") {
      ans += 1;
    } else if (control[i] === "s") {
      ans -= 1;
    } else if (control[i] === "d") {
      ans += 10;
    } else {
      ans -= 10;
    }
  }
  return ans;
}

const operations = {
  w: (n) => n + 1,
  s: (n) => n - 1,
  d: (n) => n + 10,
  a: (n) => n - 10,
};

function solution(n, control) {
  return [...control].reduce((prev, op) => operations[op](prev), n);
}
