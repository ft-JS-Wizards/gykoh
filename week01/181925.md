```javascript
function solution(numLog) {
  let ans = "";
  for (let i = 1; i < numLog.length; i++) {
    if (numLog[i] - numLog[i - 1] === 1) {
      ans += "w";
    } else if (numLog[i] - numLog[i - 1] === -1) {
      ans += "s";
    } else if (numLog[i] - numLog[i - 1] === 10) {
      ans += "d";
    } else {
      ans += "a";
    }
  }
  return ans;
}
```

```javascript
function solution(numLog) {
  const convert = {
    1: "w",
    "-1": "s",
    10: "d",
    "-10": "a",
  };

  return numLog
    .slice(1)
    .map((v, i) => {
      return convert[v - numLog[i]];
    })
    .join("");
}
```
