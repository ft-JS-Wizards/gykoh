function solution(num_list) {

  let oddNum = 0;
  for (let i = 0; i < num_list.length; i++) {
    num_list[i] % 2

function solution(num_list) {
  const { odds, evens } = num_list.reduce(
    ({ odds, evens }, num) => {
      if (num % 2 === 0) {
        evens.push(num);
      } else {
        odds.push(num);
      }
      return { odds, evens };
    },
    { odds: [], evens: [] }
  );
  return Number(odds.join("")) + Number(evens.join(""));
}
